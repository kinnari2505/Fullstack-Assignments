{
    counter = (counter - 1 + images.length) % images.length;
    updateSlider();
});

function updateSlider() 
{
    const size = slider.clientWidth;
    slider.style.transform = `translateX(${-size * counter}px)`;
}
//slider.clientWidth :- finds the current width of the visible slider area,responsive to window size.
//translateX(-size * counter) :- Shifts the slider to the left by:
//0 px for first image (counter = 0)

//size px for second image (counter = 1)

//size * 2 px for third image (counter = 2),This creates the sliding effect.
